    \begin{equation*} D = \begin{bmatrix} 2 \amp 0 \\ 0 \amp 1 \\ \end{bmatrix}, \hspace{24pt} P = \begin{bmatrix} 0 \amp 1 \\ 1 \amp 1 \\ \end{bmatrix}\text{.} \end{equation*}
