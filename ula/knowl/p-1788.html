\begin{equation*} T\left(\twovec{x_1}{x_2}\right) = \begin{bmatrix} 3 \amp -2 \\ 1 \amp 2 \\ 0 \amp 3 \end{bmatrix} \twovec{x_1}{x_2} = \cthreevec{3x_1 - 2x_2}{x_1 + 2x_2}{3x_2}. \end{equation*}
