  1. \(\xvec_3 = A\xvec_2\text{.}\)

  2. \(\xvec_4 = A\xvec_3\text{.}\)

  3. \(\xvec_5 = A\xvec_4\text{.}\)

  4. \(\xvec_6 = A\xvec_5\text{.}\)
